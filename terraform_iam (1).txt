# terraform/iam.tf
#
# PURPOSE: Define IAM roles and policies for secure AWS resource access
#
# WHY THIS FILE EXISTS:
# - AWS requires explicit permissions for everything (security best practice)
# - Implements "least privilege" - each service gets only what it needs
# - Separates permissions by function (telemetry processing vs API access)
#
# KEY CONCEPT: IAM Role = "job description" for AWS services
# IAM Policy = "list of allowed actions" attached to a role

# ==============================================================================
# LAMBDA EXECUTION ROLE FOR TELEMETRY PROCESSING
# ==============================================================================

# IAM Role: What Lambda can pretend to be
# WHY: Lambda needs permission to call other AWS services
resource "aws_iam_role" "lambda_telemetry_role" {
  name               = "${local.name_prefix}-lambda-telemetry-role"
  description        = "Role for Lambda function processing EV station telemetry"
  
  # Trust policy: WHO can assume this role
  # ANSWER: Only Lambda service (lambda.amazonaws.com)
  assume_role_policy = jsonencode({
    Version = "2012-10-17"
    Statement = [
      {
        Effect = "Allow"
        Principal = {
          Service = "lambda.amazonaws.com"
        }
        Action = "sts:AssumeRole"
      }
    ]
  })

  tags = local.common_tags
}

# Attach AWS managed policy for basic Lambda execution
# WHY: Allows Lambda to write logs to CloudWatch (standard for all Lambdas)
resource "aws_iam_role_policy_attachment" "lambda_basic_execution" {
  role       = aws_iam_role.lambda_telemetry_role.name
  policy_arn = "arn:aws:iam::aws:policy/service-role/AWSLambdaBasicExecutionRole"
  # This managed policy includes:
  # - logs:CreateLogGroup
  # - logs:CreateLogStream
  # - logs:PutLogEvents
}

# Custom inline policy: WHAT this Lambda can do
# WHY: Grant specific permissions needed for telemetry processing
resource "aws_iam_role_policy" "lambda_telemetry_policy" {
  name = "${local.name_prefix}-lambda-telemetry-policy"
  role = aws_iam_role.lambda_telemetry_role.id

  policy = jsonencode({
    Version = "2012-10-17"
    Statement = [
      {
        # Permission: Write to DynamoDB
        # WHY: Lambda stores latest station state here
        Effect = "Allow"
        Action = [
          "dynamodb:PutItem",      # Create/update items
          "dynamodb:UpdateItem",   # Update existing items
          "dynamodb:GetItem"       # Read items (optional, for checks)
        ]
        Resource = aws_dynamodb_table.stations.arn
      },
      {
        # Permission: Write to S3
        # WHY: Lambda archives raw telemetry as JSON files
        Effect = "Allow"
        Action = [
          "s3:PutObject",          # Upload files
          "s3:PutObjectAcl"        # Set file permissions
        ]
        Resource = "${aws_s3_bucket.telemetry_data.arn}/*"
        # Note: "/*" means all objects in bucket, not the bucket itself
      },
      {
        # Permission: Read S3 bucket properties (optional)
        # WHY: May be needed to check bucket configuration
        Effect = "Allow"
        Action = [
          "s3:ListBucket",
          "s3:GetBucketLocation"
        ]
        Resource = aws_s3_bucket.telemetry_data.arn
      }
    ]
  })
}

# ==============================================================================
# LAMBDA EXECUTION ROLE FOR API HANDLER
# ==============================================================================

# Separate role for API Lambda (different permissions)
# WHY: API only needs to READ from DynamoDB, not write
resource "aws_iam_role" "lambda_api_role" {
  name               = "${local.name_prefix}-lambda-api-role"
  description        = "Role for Lambda function handling API requests"
  
  # Same trust policy - Lambda service can assume this role
  assume_role_policy = jsonencode({
    Version = "2012-10-17"
    Statement = [
      {
        Effect = "Allow"
        Principal = {
          Service = "lambda.amazonaws.com"
        }
        Action = "sts:AssumeRole"
      }
    ]
  })

  tags = local.common_tags
}

# Attach basic execution policy (CloudWatch logs)
resource "aws_iam_role_policy_attachment" "lambda_api_basic_execution" {
  role       = aws_iam_role.lambda_api_role.name
  policy_arn = "arn:aws:iam::aws:policy/service-role/AWSLambdaBasicExecutionRole"
}

# Custom policy for API Lambda
# WHY: Only needs READ access to DynamoDB
resource "aws_iam_role_policy" "lambda_api_policy" {
  name = "${local.name_prefix}-lambda-api-policy"
  role = aws_iam_role.lambda_api_role.id

  policy = jsonencode({
    Version = "2012-10-17"
    Statement = [
      {
        # Permission: Read from DynamoDB
        # WHY: API retrieves current station status
        Effect = "Allow"
        Action = [
          "dynamodb:GetItem",      # Get single item by key
          "dynamodb:Query",        # Query with conditions
          "dynamodb:Scan"          # Scan entire table (use sparingly)
        ]
        Resource = aws_dynamodb_table.stations.arn
      }
    ]
  })
}

# IOT CORE ROLE FOR TRIGGERING LAMBDA

  name               = "${local.name_prefix}-iot-lambda-role"
  description        = "Role for IoT Core to invoke Lambda functions"
  
  # Trust policy: IoT Core service can assume this role
  assume_role_policy = jsonencode({
    Version = "2012-10-17"
    Statement = [
      {
        Effect = "Allow"
        Principal = {
          Service = "iot.amazonaws.com"
        }
        Action = "sts:AssumeRole"
      }
    ]
  })

  tags = local.common_tags
}

# Policy allowing IoT to invoke Lambda
# WHY: IoT Rule needs to call Lambda when messages arrive
resource "aws_iam_role_policy" "iot_lambda_invoke_policy" {
  name = "${local.name_prefix}-iot-lambda-invoke"
  role = aws_iam_role.iot_lambda_role.id

  policy = jsonencode({
    Version = "2012-10-17"
    Statement = [
      {
        Effect = "Allow"
        Action = [
          "lambda:InvokeFunction"  # Call Lambda function
        ]
        Resource = aws_lambda_function.telemetry_handler.arn
      }
    ]
  })
}
