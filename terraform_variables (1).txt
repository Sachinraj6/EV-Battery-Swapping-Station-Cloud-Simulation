# terraform/variables.tf
#
# PURPOSE: Define all configurable parameters for the infrastructure
#
# WHY THIS FILE EXISTS:
# - Makes the infrastructure flexible and reusable
# - Allows different environments (dev, prod) without code changes
# - Documents what can be configured
# - Provides sensible defaults for learning/prototype use

# Project name used in resource naming
# WHY: Groups related resources together
variable "project_name" {
  description = "Name of the project, used for resource naming and tagging"
  type        = string
  default     = "ev-swap"
  
  # Validation ensures we follow AWS naming conventions
  validation {
    condition     = can(regex("^[a-z0-9-]+$", var.project_name))
    error_message = "Project name must contain only lowercase letters, numbers, and hyphens."
  }
}

# Environment (dev, staging, prod)
# WHY: Allows separate infrastructure for different stages
variable "environment" {
  description = "Environment name (dev, staging, prod)"
  type        = string
  default     = "dev"
  
  validation {
    condition     = contains(["dev", "staging", "prod"], var.environment)
    error_message = "Environment must be dev, staging, or prod."
  }
}

# AWS Region where resources will be created
# WHY: Some services are region-specific, costs vary by region
variable "aws_region" {
  description = "AWS region where resources will be created"
  type        = string
  default     = "us-east-1"  # N. Virginia - generally cheapest region
}

# Number of stations to configure for
# WHY: Helps estimate costs and set appropriate limits
variable "station_count" {
  description = "Expected number of battery swap stations (for cost estimation)"
  type        = number
  default     = 20  # Realistic for prototype/demo
  
  validation {
    condition     = var.station_count > 0 && var.station_count <= 100
    error_message = "Station count must be between 1 and 100 for this prototype."
  }
}

# Lambda function timeout in seconds
# WHY: Prevents runaway functions from costing money
variable "lambda_timeout" {
  description = "Timeout for Lambda functions in seconds"
  type        = number
  default     = 30  # Telemetry processing should be fast
  
  validation {
    condition     = var.lambda_timeout >= 3 && var.lambda_timeout <= 900
    error_message = "Lambda timeout must be between 3 and 900 seconds."
  }
}

# Lambda function memory allocation in MB
# WHY: More memory = more CPU power = faster execution
# COST TRADEOFF: More memory costs more per execution
variable "lambda_memory_size" {
  description = "Memory allocation for Lambda functions in MB"
  type        = number
  default     = 256  # Enough for basic JSON processing
  
  validation {
    condition     = var.lambda_memory_size >= 128 && var.lambda_memory_size <= 10240
    error_message = "Lambda memory must be between 128 and 10240 MB."
  }
}

# Python runtime version for Lambda
# WHY: Specifies which Python version Lambda will use
variable "lambda_runtime" {
  description = "Python runtime version for Lambda functions"
  type        = string
  default     = "python3.11"
  
  validation {
    condition     = can(regex("^python3\\.(9|10|11|12)$", var.lambda_runtime))
    error_message = "Lambda runtime must be python3.9, python3.10, python3.11, or python3.12."
  }
}

# DynamoDB billing mode
# WHY: On-demand is great for unpredictable workloads, provisioned for steady loads
# COST: On-demand = pay per request, Provisioned = pay for capacity
variable "dynamodb_billing_mode" {
  description = "DynamoDB billing mode (PAY_PER_REQUEST or PROVISIONED)"
  type        = string
  default     = "PAY_PER_REQUEST"  # Best for learning - no capacity planning needed
  
  validation {
    condition     = contains(["PAY_PER_REQUEST", "PROVISIONED"], var.dynamodb_billing_mode)
    error_message = "Billing mode must be PAY_PER_REQUEST or PROVISIONED."
  }
}

# S3 lifecycle policy - move old data to cheaper storage
# WHY: Raw telemetry gets expensive over time, glacier is much cheaper
variable "s3_glacier_transition_days" {
  description = "Days before transitioning S3 objects to Glacier (archival storage)"
  type        = number
  default     = 90  # Keep recent data in S3, archive older data
  
  validation {
    condition     = var.s3_glacier_transition_days >= 30
    error_message = "Glacier transition must be at least 30 days (AWS minimum)."
  }
}

# CloudWatch log retention
# WHY: Logs cost money to store, old logs rarely needed
variable "cloudwatch_log_retention_days" {
  description = "Days to retain CloudWatch logs"
  type        = number
  default     = 7  # One week is enough for debugging
  
  validation {
    condition     = contains([1, 3, 5, 7, 14, 30, 60, 90, 120, 150, 180, 365, 400, 545, 731, 1827, 3653], var.cloudwatch_log_retention_days)
    error_message = "Log retention must be an AWS-supported value."
  }
}

# Enable detailed CloudWatch metrics
# WHY: More metrics = better visibility, but also more cost
variable "enable_detailed_monitoring" {
  description = "Enable detailed CloudWatch monitoring (costs extra)"
  type        = bool
  default     = false  # Keep costs down for prototype
}

# IoT topic name pattern
# WHY: Defines how stations organize their messages
variable "iot_topic_prefix" {
  description = "MQTT topic prefix for station telemetry"
  type        = string
  default     = "ev/station"
  
  # Topic will be: ev/station/{station_id}/telemetry
}
